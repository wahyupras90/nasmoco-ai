CONTOH
════════════════════════════════════════

User:
revenue SAID juni 2026

SQL:

SELECT
sa,
SUM(revenue) AS revenue
FROM daily_kpi
WHERE sa='SAID'
AND is_counter=0
AND strftime('%Y-%m',tanggal)='2026-06'
GROUP BY sa;

────────────────────────

User:
ranking revenue SA juni 2026

SQL:

SELECT
sa,
SUM(revenue) AS revenue
FROM daily_kpi
WHERE is_counter=0
AND strftime('%Y-%m',tanggal)='2026-06'
GROUP BY sa
ORDER BY revenue DESC;

────────────────────────

User:
pencapaian cpus dan revenue vs target di mei 2026

SQL:

SELECT
d.sa,
SUM(d.cpus) AS actual_cpus,
t.target_cpus,
ROUND(
SUM(d.cpus)*100.0/
NULLIF(t.target_cpus,0),
1
) AS cpus_pct,
SUM(d.revenue) AS actual_revenue,
t.target_revenue,
ROUND(
SUM(d.revenue)*100.0/
NULLIF(t.target_revenue,0),
1
) AS revenue_pct
FROM daily_kpi d
JOIN target_bulanan t
ON d.sa=t.sa
AND t.tahun=2026
AND t.bulan=5
WHERE d.is_counter=0
AND strftime('%Y-%m',d.tanggal)='2026-05'
GROUP BY d.sa;
